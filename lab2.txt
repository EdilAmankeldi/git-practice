Задача 1. Минимум из двух чисел
a = int(input("Введите первое число: "))
b = int(input("Введите второе число: "))

if a < b:
    print("Меньшее число:", a)
else:
    print("Меньшее число:", b)

Задача 2. Четное или нечетное
n = int(input("Введите число: "))

if n % 2 == 0:
    print("Число чётное")
else:
    print("Число нечётное")

Задача 3. Сумма чисел от 1 до N
n = int(input("Введите N: "))
s = 0

for i in range(1, n+1):
    s += i

print("Сумма чисел от 1 до", n, "=", s)

Задача 4. Таблица умножения
n = int(input("Введите число N: "))

for i in range(1, 11):
    print(n, "*", i, "=", n * i)

Задача 5. Факториал числа
n = int(input("Введите число N: "))
f = 1

for i in range(1, n+1):
    f *= i

print("Факториал числа", n, "=", f)

Задача 6. Количество гласных
s = input("Введите строку: ").lower()
vowels = "аеёиоуыэюя"
count = 0

for ch in s:
    if ch in vowels:
        count += 1

print("Количество гласных:", count)

Задача 7. Разворот строки
s = input("Введите строку: ")
print("Обратная строка:", s[::-1])

Задача 8. Максимум в массиве
arr = list(map(int, input("Введите массив чисел через пробел: ").split()))
print("Максимальное число:", max(arr))

Задача 9. Сумма элементов массива
arr = list(map(int, input("Введите массив чисел через пробел: ").split()))
print("Сумма элементов массива:", sum(arr))

Задача 10. Поиск элемента в массиве
arr = list(map(int, input("Введите массив чисел через пробел: ").split()))
x = int(input("Введите число X: "))

if x in arr:
    print("Число", x, "содержится в массиве")
else:
    print("Число", x, "не найдено")
